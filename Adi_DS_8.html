# (A) Countplot - Number of Passengers by Class

plt.figure(figsize=(6,4))  # Set figure size
sns.countplot(x='class', data=titanic, palette='pastel')  # Create countplot
plt.title('Number of Passengers in Each Class')  # Add title

In[9]:
# (B) Gender Distribution of Survivors

plt.figure(figsize=(6,4))
sns.countplot(x='sex', hue='survived', data=titanic, palette='Set2')
plt.title('Survival Count by Gender')
plt.legend(title="Survived", labels=["No", "Yes"])
plt.show()
In[10]:
# (C) Boxplot  Ticket Fare by Passenger Class

plt.figure(figsize=(8,5))  # Set figure size
sns.boxplot(x='class', y='fare', data=titanic, palette='coolwarm')  
plt.title('Ticket Fare Distribution by Passenger Class')
plt.xlabel('Passenger Class')
plt.ylabel('Fare ($)')
plt.show()
In[11]:
# (D) Bar Chart  Survival Rate by Passenger Class

plt.figure(figsize=(6,4))
sns.barplot(x='class', y='survived', data=titanic, palette='viridis', estimator=lambda x: sum(x)/len(x))
plt.title('Survival Rate by Passenger Class')
plt.ylabel('Survival Rate')
plt.xlabel('Passenger Class')
plt.ylim(0, 1)  # Scale from 0 to 1 (percentage)
plt.show()
